package com.carhire.service;
import com.carhire.model.Booking;
import com.carhire.model.Customer;
import com.carhire.model.Vehicle;
import com.carhire.model.VehicleCategory;
import org.junit.jupiter.api.AfterEach;
import org.junit.jupiter.api.BeforeEach;
import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.*;

public class BookingServiceTest {
private BookingService bookingService;

@BeforeEach
    void setUp(){
    bookingService = new BookingServiceImpl();
}
@Test
    void testValidBooking_StandardCar_ShouldreturnTrue(){
    Customer customer = new Customer("1","John",30,10);
    Vehicle vehicle = new Vehicle("ABC123",VehicleCategory.STANDARD,50.0);
    Booking booking = new Booking(customer, vehicle, 5);
    boolean isValid = bookingService.isValidBooking(booking);
    assertTrue(isValid,"vzroslii i obichnoe auto");
}
@Test
    void testInvalidBooking_UnderrageCustomer_shouldReturnFalse(){
    Customer underageCustomer = new Customer("2","Anton",17,0);
    Vehicle vehicle = new Vehicle("X999Y",VehicleCategory.ECONOMY,30.0);
    Booking booking = new Booking(underageCustomer, vehicle, 3);
    boolean isValid = bookingService.isValidBooking(booking);
    assertFalse(isValid,"malo godikov");
}
@Test
    void testInvalidBooking_PremiumCarRequirementsNotMet_shouldReturnFalse(){
    Customer customer = new Customer("3","Sin dodona",23,2);
    Vehicle premiumVehicle = new Vehicle("PID12",VehicleCategory.PREMIUM,150.0);
    Booking booking = new Booking(customer, premiumVehicle, 10);
    boolean isValid = bookingService.isValidBooking(booking);
    assertFalse(isValid,"idi nahui");

}
}
